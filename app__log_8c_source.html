64 /*
65  * Store given log record to flash. Keeps track of slots via state variables.
66  */
67 static rd_status_t store_block (const app_log_record_t * const p_record)
68 {
69  rd_status_t err_code = RD_SUCCESS;
70  uint8_t num_tries = 0;
71  static uint8_t record_idx = 0;
72 
73  do
74  {
75  // If loop went past the last record, wrap.
76  uint8_t record_slot = (record_idx + num_tries) % APP_FLASH_LOG_DATA_RECORDS_NUM;
77  uint16_t target_record = (APP_FLASH_LOG_DATA_RECORD_PREFIX << 8u) + record_slot;
78  // Free space for new record if there already is one.
79  err_code = rt_flash_free (APP_FLASH_LOG_FILE, target_record);
80  char msg[128];
81  snprintf (msg, sizeof (msg), "Storing logs in record #%04X.\n", target_record);
82  LOGD (msg);
83  // Clear out error if there was no record to erase out of way.
84  err_code &= ~RD_ERROR_NOT_FOUND;
85 
86  while (rt_flash_busy()) { ri_yield(); }
87 
88  // Run GC to actually release the space of old record.
89  err_code |= rt_flash_gc_run ();
90 
91  while (rt_flash_busy()) { ri_yield(); }
92 
93  err_code |= rt_flash_store (APP_FLASH_LOG_FILE, target_record,
94  p_record, sizeof (app_log_record_t));
95 
96  while (rt_flash_busy()) { ri_yield(); }
97 
98  RD_ERROR_CHECK (err_code, ~RD_ERROR_FATAL);
99  num_tries++; // Try the next block if there was error.
100  } while ( (RD_SUCCESS != err_code) && (num_tries < APP_FLASH_LOG_DATA_RECORDS_NUM));
101 
102  if (RD_SUCCESS == err_code)
103  {
104  // Increment the record to the slot that was successful
105  record_idx += num_tries;
106  record_idx = record_idx % APP_FLASH_LOG_DATA_RECORDS_NUM;
107  }
108 
109  return err_code;
110 }

112 static rd_status_t purge_logs (void)
113 {
114  rd_status_t err_code = RD_SUCCESS;
115  LOGI ("Free all & GC\n");
116 
117  for (uint8_t record_idx = 0; record_idx < APP_FLASH_LOG_DATA_RECORDS_NUM; record_idx++)
118  {
119  err_code |= rt_flash_free (APP_FLASH_LOG_FILE,
120  (APP_FLASH_LOG_DATA_RECORD_PREFIX << 8u) + record_idx);
121 
122  while (rt_flash_busy()) { ri_yield(); }
123  }
124 
125  err_code &= ~RD_ERROR_NOT_FOUND; // It doesn't matter if there was no data to erase.
126  err_code |= rt_flash_gc_run ();
127  return err_code;
128 }

187 rd_status_t app_log_process (const rd_sensor_data_t * const sample)
188 {
189  rd_status_t err_code = RD_SUCCESS;
190  uint64_t next_sample_ms = m_last_sample_ms + (m_log_config.interval_s * 1000u);
191  uint32_t end_timestamp = sample->timestamp_ms / 1000U;
192  LOGD ("Sample received\n");
193 
194  if (0 == m_last_sample_ms) { next_sample_ms = 0; } // Always store first sample.
195 
196  if (next_sample_ms <= sample->timestamp_ms) //Check if new sample should be processed
197  {
198  LOGD ("Storing sample\n");
199  app_log_element_t element =
200  {
201  .timestamp_s = sample->timestamp_ms / 1000u,
202  .temperature_c = rd_sensor_data_parse (sample, RD_SENSOR_TEMP_FIELD),
203  .humidity_rh = rd_sensor_data_parse (sample, RD_SENSOR_HUMI_FIELD),
204  .pressure_pa = rd_sensor_data_parse (sample, RD_SENSOR_PRES_FIELD),
205  };
206 
207  if (APP_LOG_MAX_SAMPLES > m_log_input_block.num_samples)
208  { m_log_input_block.storage[m_log_input_block.num_samples++] = element; }
209 
210  if (m_log_input_block.num_samples >= APP_LOG_MAX_SAMPLES)
211  {
212  LOGI ("Storing block\n");
213  m_log_input_block.end_timestamp_s = end_timestamp;
214  err_code |= store_block (&m_log_input_block);
215  RD_ERROR_CHECK (err_code, RD_SUCCESS);
216  memset (&m_log_input_block, 0, sizeof (m_log_input_block)); // zero input_block
217  m_log_input_block.start_timestamp_s = end_timestamp;
218  }
219 
220  m_last_sample_ms = sample->timestamp_ms;
221  }
222 
223  return err_code;
224 }

232 static rd_status_t app_log_read_load_block (app_log_read_state_t * const p_rs)
233 {
234  rd_status_t err_code = RD_SUCCESS;
235 
236  if ( (0 == p_rs->element_idx)
237  && (0 == p_rs->page_idx))
238  {
239  // Clear out previous state
240  err_code |= rt_flash_load (APP_FLASH_LOG_FILE,
241  (APP_FLASH_LOG_DATA_RECORD_PREFIX << 8u) + p_rs->page_idx,
242  &m_log_output_block, sizeof (m_log_output_block));
243  p_rs->page_idx++;
244  }
245  else if ( (APP_FLASH_LOG_DATA_RECORDS_NUM > p_rs->page_idx)
246  && (p_rs->element_idx >= m_log_output_block.num_samples))
247  {
248  // Returns NOT_FOUND if page IDX is not in flash.
249  err_code |= rt_flash_load (APP_FLASH_LOG_FILE,
250  (APP_FLASH_LOG_DATA_RECORD_PREFIX << 8u) + p_rs->page_idx,
251  &m_log_output_block, sizeof (m_log_output_block));
252  p_rs->page_idx++;
253  p_rs->element_idx = 0;
254  }
255  else if ( (APP_FLASH_LOG_DATA_RECORDS_NUM == p_rs->page_idx)
256  && (p_rs->element_idx >= m_log_output_block.num_samples))
257  {
258  memcpy (&m_log_output_block, &m_log_input_block, sizeof (m_log_output_block));
259  p_rs->page_idx++;
260  p_rs->element_idx = 0;
261  }
262  else {} // No action needed.
263 
264  // Zero out state if block was not found
265  if (RD_ERROR_NOT_FOUND == err_code)
266  {
267  memset (&m_log_output_block, 0, sizeof (m_log_output_block));
268  }
269 
270  return err_code;
271 }
