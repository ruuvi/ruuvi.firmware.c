63/*
64 * Store given log record to flash. Keeps track of slots via state variables.
65 */
66static rd_status_t store_block (const app_log_record_t * const p_record)
67{
68 rd_status_t err_code = RD_SUCCESS;
69 uint8_t num_tries = 0;
70 static uint8_t record_idx = 0;
71
72 do
73 {
74 // If loop went past the last record, wrap.
75 uint8_t record_slot = (record_idx + num_tries) % APP_FLASH_LOG_DATA_RECORDS_NUM;
76 uint16_t target_record = (APP_FLASH_LOG_DATA_RECORD_PREFIX << 8u) + record_slot;
77 // Free space for new record if there already is one.
78 err_code = rt_flash_free (APP_FLASH_LOG_FILE, target_record);
79 char msg[128];
80 snprintf (msg, sizeof (msg), "Storing logs in record #%04X.\n", target_record);
81 LOGD (msg);
82 // Clear out error if there was no record to erase out of way.
83 err_code &= ~RD_ERROR_NOT_FOUND;
84
85 while (rt_flash_busy()) { ri_yield(); }
86
87 // Run GC to actually release the space of old record.
88 err_code |= rt_flash_gc_run ();
89
90 while (rt_flash_busy()) { ri_yield(); }
91
92 err_code |= rt_flash_store (APP_FLASH_LOG_FILE, target_record,
93 p_record, sizeof (app_log_record_t));
94
95 while (rt_flash_busy()) { ri_yield(); }
96
97 RD_ERROR_CHECK (err_code, ~RD_ERROR_FATAL);
98 num_tries++; // Try the next block if there was error.
99 } while ( (RD_SUCCESS != err_code) && (num_tries < APP_FLASH_LOG_DATA_RECORDS_NUM));
100
101 if (RD_SUCCESS == err_code)
102 {
103 // Increment the record to the slot that was successful
104 record_idx += num_tries;
105 record_idx = record_idx % APP_FLASH_LOG_DATA_RECORDS_NUM;
106 }
107
108 return err_code;
109}

111static rd_status_t purge_logs (void)
112{
113 rd_status_t err_code = RD_SUCCESS;
114 LOGI ("Free all & GC\n");
115
116 for (uint8_t record_idx = 0; record_idx < APP_FLASH_LOG_DATA_RECORDS_NUM; record_idx++)
117 {
118 err_code |= rt_flash_free (APP_FLASH_LOG_FILE,
119 (APP_FLASH_LOG_DATA_RECORD_PREFIX << 8u) + record_idx);
120
121 while (rt_flash_busy()) { ri_yield(); }
122 }
123
124 err_code &= ~RD_ERROR_NOT_FOUND; // It doesn't matter if there was no data to erase.
125 err_code |= rt_flash_gc_run ();
126 return err_code;
127}

192rd_status_t app_log_process (const rd_sensor_data_t * const sample)
193{
194 rd_status_t err_code = RD_SUCCESS;
195 uint64_t next_sample_ms = m_last_sample_ms + (m_log_config.interval_s * 1000u);
196 uint32_t end_timestamp = sample->timestamp_ms / 1000U;
197 LOGD ("Sample received\n");
198
199 if (0 == m_last_sample_ms) { next_sample_ms = 0; } // Always store first sample.
200
201 if (next_sample_ms <= sample->timestamp_ms) //Check if new sample should be processed
202 {
203 LOGD ("Storing sample\n");
204 app_log_element_t element =
205 {
206 .timestamp_s = sample->timestamp_ms / 1000u,
207 .temperature_c = rd_sensor_data_parse (sample, RD_SENSOR_TEMP_FIELD),
208 .humidity_rh = rd_sensor_data_parse (sample, RD_SENSOR_HUMI_FIELD),
209 .pressure_pa = rd_sensor_data_parse (sample, RD_SENSOR_PRES_FIELD),
210 };
211
212 if (APP_LOG_MAX_SAMPLES > m_log_input_block.num_samples)
213 { m_log_input_block.storage[m_log_input_block.num_samples++] = element; }
214
215 if (m_log_input_block.num_samples >= APP_LOG_MAX_SAMPLES)
216 {
217 LOGI ("Storing block\n");
218 m_log_input_block.end_timestamp_s = end_timestamp;
219 err_code |= store_block (&m_log_input_block);
220 RD_ERROR_CHECK (err_code, RD_SUCCESS);
221 memset (&m_log_input_block, 0, sizeof (m_log_input_block)); // zero input_block
222 m_log_input_block.start_timestamp_s = end_timestamp;
223 }
224
225 m_last_sample_ms = sample->timestamp_ms;
226 }
227
228 return err_code;
229}

237static rd_status_t app_log_read_load_block (app_log_read_state_t * const p_rs)
238{
239 rd_status_t err_code = RD_SUCCESS;
240
241 if ( (0 == p_rs->element_idx)
242 && (0 == p_rs->page_idx))
243 {
244 // Clear out previous state
245 err_code |= rt_flash_load (APP_FLASH_LOG_FILE,
246 (APP_FLASH_LOG_DATA_RECORD_PREFIX << 8u) + p_rs->page_idx,
247 &m_log_output_block, sizeof (m_log_output_block));
248 p_rs->page_idx++;
249 }
250 else if ( (APP_FLASH_LOG_DATA_RECORDS_NUM > p_rs->page_idx)
251 && (p_rs->element_idx >= m_log_output_block.num_samples))
252 {
253 // Returns NOT_FOUND if page IDX is not in flash.
254 err_code |= rt_flash_load (APP_FLASH_LOG_FILE,
255 (APP_FLASH_LOG_DATA_RECORD_PREFIX << 8u) + p_rs->page_idx,
256 &m_log_output_block, sizeof (m_log_output_block));
257 p_rs->page_idx++;
258 p_rs->element_idx = 0;
259 }
260 else if ( (APP_FLASH_LOG_DATA_RECORDS_NUM == p_rs->page_idx)
261 && (p_rs->element_idx >= m_log_output_block.num_samples))
262 {
263 memcpy (&m_log_output_block, &m_log_input_block, sizeof (m_log_output_block));
264 p_rs->page_idx++;
265 p_rs->element_idx = 0;
266 }
267 else {} // No action needed.
268
269 // Zero out state if block was not found
270 if (RD_ERROR_NOT_FOUND == err_code)
271 {
272 memset (&m_log_output_block, 0, sizeof (m_log_output_block));
273 }
274
275 return err_code;
276}

285static rd_status_t app_log_read_fast_forward (app_log_read_state_t * const p_rs)
286{
287 rd_status_t err_code = RD_SUCCESS;
288 uint64_t ts_s = m_log_output_block.storage[p_rs->element_idx].timestamp_s;
289 uint64_t ts_ms = ts_s * 1000LLU;
290
291 while ( (p_rs->oldest_element_ms > ts_ms)
292 && (p_rs->element_idx < m_log_output_block.num_samples))
293 {
294 ts_s = m_log_output_block.storage[p_rs->element_idx].timestamp_s;
295 ts_ms = ts_s * 1000LLU;
296 p_rs->element_idx++;
297 }
298
299 if (p_rs->element_idx >= m_log_output_block.num_samples) { err_code |= RD_ERROR_NOT_FOUND; }
300
301 return err_code;
302}

304static rd_status_t app_log_read_populate (rd_sensor_data_t * const sample,
305 app_log_read_state_t * const p_rs)
306{
307 rd_status_t err_code = RD_SUCCESS;
308
309 // Last memory block is the current RAM buffer, so we have valid data
310 // when APP_FLASH_LOG_DATA_RECORDS_NUM + 1 == p_rs->page_idx
311 if ( (APP_FLASH_LOG_DATA_RECORDS_NUM + 1) < p_rs->page_idx)
312 {
313 err_code |= RD_ERROR_NOT_FOUND;
314 }
315 else if (p_rs->element_idx < m_log_output_block.num_samples)
316 {
317 const app_log_element_t * const p_el = &m_log_output_block.storage[p_rs->element_idx];
318 rd_sensor_data_set (sample, RD_SENSOR_TEMP_FIELD, p_el->temperature_c);
319 rd_sensor_data_set (sample, RD_SENSOR_HUMI_FIELD, p_el->humidity_rh);
320 rd_sensor_data_set (sample, RD_SENSOR_PRES_FIELD, p_el->pressure_pa);
321 sample->timestamp_ms = ( (uint64_t) (p_el->timestamp_s)) * 1000LLU;
322 p_rs->element_idx++;
323 }
324 else {} // No action required.
325
326 return err_code;
327}

356rd_status_t app_log_config_set (const app_log_config_t * const configuration)
357{
358 rd_status_t err_code = RD_SUCCESS;
359 uint32_t end_timestamp = m_log_input_block.end_timestamp_s;
360
361 if (NULL == configuration) { err_code |= RD_ERROR_NULL; }
362 else
363 {
364 err_code |= rt_flash_store (APP_FLASH_LOG_FILE, APP_FLASH_LOG_CONFIG_RECORD,
365 &configuration, sizeof (configuration));
366
367 if (RD_SUCCESS == err_code)
368 {
369 err_code |= store_block (&m_log_input_block);
370 RD_ERROR_CHECK (err_code, RD_SUCCESS);
371 memset (&m_log_input_block, 0, sizeof (m_log_input_block));
372 m_log_input_block.start_timestamp_s = end_timestamp;
373 memcpy (&m_log_config, configuration, sizeof (m_log_config));
374 }
375 }
376
377 return err_code;
