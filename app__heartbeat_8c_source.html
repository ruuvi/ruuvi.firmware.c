86 void heartbeat (void * p_event, uint16_t event_size)
87 {
88  ri_comm_message_t msg = {0};
89  rd_status_t err_code = RD_SUCCESS;
90  bool heartbeat_ok = false;
91  rd_sensor_data_t data = { 0 };
92  size_t buffer_len = RI_COMM_MESSAGE_MAX_LENGTH;
93  data.fields = app_sensor_available_data();
94  float data_values[rd_sensor_data_fieldcount (&data)];
95  data.data = data_values;
96  app_sensor_get (&data);
97  // Sensor read takes a long while, indicate activity once data is read.
98  app_led_activity_signal (true);
99  m_dataformat_state = app_dataformat_next (m_dataformats_enabled, m_dataformat_state);
100  app_dataformat_encode (msg.data, &buffer_len, &data, m_dataformat_state);
101  msg.data_length = (uint8_t) buffer_len;
102  err_code = send_adv (&msg);
103  // Advertising should always be successful
104  RD_ERROR_CHECK (err_code, ~RD_ERROR_FATAL);
105 
106  if (RD_SUCCESS == err_code)
107  {
108  heartbeat_ok = true;
109  }
110 
111  // Cut endpoint data to fit into GATT msg.
112  msg.data_length = 18;
113  // Gatt Link layer takes care of delivery.
114  msg.repeat_count = 1;
115  err_code = rt_gatt_send_asynchronous (&msg);
116 
117  if (RD_SUCCESS == err_code)
118  {
119  heartbeat_ok = true;
120  }
121 
122  // Restore original message length for NFC
123  msg.data_length = (uint8_t) buffer_len;
124  err_code = rt_nfc_send (&msg);
125 
126  if (RD_SUCCESS == err_code)
127  {
128  heartbeat_ok = true;
129  }
130 
131  if (heartbeat_ok)
132  {
133  ri_watchdog_feed();
134  last_heartbeat_timestamp_ms = ri_rtc_millis();
135  }
136 
137  // Turn LED off before starting lengthy flash operations
138  app_led_activity_signal (false);
139  err_code = app_log_process (&data);
140  RD_ERROR_CHECK (err_code, ~RD_ERROR_FATAL);
141 }
