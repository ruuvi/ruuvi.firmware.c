83 void heartbeat (void * p_event, uint16_t event_size)
84 {
85  ri_comm_message_t msg = {0};
86  rd_status_t err_code = RD_SUCCESS;
87  bool heartbeat_ok = false;
88  rd_sensor_data_t data = { 0 };
89  size_t buffer_len = RI_COMM_MESSAGE_MAX_LENGTH;
90  data.fields = app_sensor_available_data();
91  float data_values[rd_sensor_data_fieldcount (&data)];
92  data.data = data_values;
93  app_sensor_get (&data);
94  // Sensor read takes a long while, indicate activity once data is read.
95  app_led_activity_signal (true);
96  m_dataformat_state = app_dataformat_next (m_dataformats_enabled, m_dataformat_state);
97  app_dataformat_encode (msg.data, &buffer_len, &data, m_dataformat_state);
98  msg.data_length = (uint8_t) buffer_len;
99  err_code = send_adv (&msg);
100  // Advertising should always be successful
101  RD_ERROR_CHECK (err_code, ~RD_ERROR_FATAL);
102 
103  if (RD_SUCCESS == err_code)
104  {
105  heartbeat_ok = true;
106  }
107 
108  // Cut endpoint data to fit into GATT msg.
109  msg.data_length = 18;
110  // Gatt Link layer takes care of delivery.
111  msg.repeat_count = 1;
112  err_code = rt_gatt_send_asynchronous (&msg);
113 
114  if (RD_SUCCESS == err_code)
115  {
116  heartbeat_ok = true;
117  }
118 
119  // Restore original message length for NFC
120  msg.data_length = (uint8_t) buffer_len;
121  err_code = rt_nfc_send (&msg);
122 
123  if (RD_SUCCESS == err_code)
124  {
125  heartbeat_ok = true;
126  }
127 
128  if (heartbeat_ok)
129  {
130  ri_watchdog_feed();
131  last_heartbeat_timestamp_ms = ri_rtc_millis();
132  }
133 
134  // Turn LED off before starting lengthy flash operations
135  app_led_activity_signal (false);
136  err_code = app_log_process (&data);
137  RD_ERROR_CHECK (err_code, ~RD_ERROR_FATAL);
138 }
