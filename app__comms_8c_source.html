135 const uint8_t * const raw_message)
136{
137 ri_comm_message_t msg = {0};
138 msg.data_length = RE_STANDARD_MESSAGE_LENGTH;
139 msg.repeat_count = 1;
140 msg.data[RE_STANDARD_DESTINATION_INDEX] = raw_message[RE_STANDARD_SOURCE_INDEX];
141 msg.data[RE_STANDARD_SOURCE_INDEX] = raw_message[RE_STANDARD_DESTINATION_INDEX];

191static rd_status_t wait_for_tx_done (const uint32_t timeout_ms)
192{
193 rd_status_t err_code = RD_SUCCESS;
194 const uint64_t start = ri_rtc_millis();
195 const uint64_t timeout = start + timeout_ms;
196
197 while ( (!m_tx_done) && (timeout > ri_rtc_millis()))
198 {
199 ri_yield();
200 }
201
202 if (!m_tx_done)
203 {
204 err_code |= RD_ERROR_TIMEOUT;
205 }
206
207 m_tx_done = false;
208 return err_code;
209}

211static rd_status_t password_check (const ri_comm_xfer_fp_t reply_fp,
212 const uint8_t * const raw_message)
213{
214 rd_status_t err_code = RD_SUCCESS;
215 uint64_t entered_password = 0U;
216 bool auth_ok = false;
217 // Use non-zero initial value so passwords won't match on error
218 uint64_t current_password = 0xDEADBEEF12345678U;
219 re_op_t op = (re_op_t) raw_message[RE_STANDARD_OPERATION_INDEX];
220
221 if (RE_STANDARD_VALUE_READ == op)
222 {
223 err_code |= ri_comm_id_get (&current_password);
224
225 // Convert data to U64
226 for (uint8_t ii = RE_STANDARD_PAYLOAD_START_INDEX;
227 ii < RE_STANDARD_MESSAGE_LENGTH; ii++)
228 {
229 uint8_t byte_offset = RE_STANDARD_MESSAGE_LENGTH - ii - 1U;
230 uint8_t bit_offset = byte_offset * 8U;
231 uint64_t password_part = raw_message[ii];
232 password_part = password_part << bit_offset;
233 entered_password += password_part;
234 }
235 }
236
237 m_tx_done = false;
238
239 if (entered_password == current_password)
240 {
241 err_code |= reply_authorized (reply_fp, raw_message);
242 auth_ok = true;
243 }
244 else
245 {
246 err_code |= reply_unauthorized (reply_fp, raw_message);
247 auth_ok = false;
248 }
249
250 err_code |= wait_for_tx_done (BLOCKING_COMM_TIMEOUT_MS);
251
252 if (auth_ok)
253 {
254 app_comms_configure_next_enable ();
255 }
256 else
257 {
258 app_comms_configure_next_disable ();
259 }
260
261 return err_code;
262}

264TESTABLE_STATIC void handle_comms (const ri_comm_xfer_fp_t reply_fp, void * p_data,
265 size_t data_len)
266{
267 rd_status_t err_code = RD_SUCCESS;
268 const uint8_t * const raw_message = (uint8_t *) p_data;
269
270 if (NULL == p_data)
271 {
272 err_code |= RD_ERROR_NULL;
273 }
274 else if (data_len < RE_STANDARD_MESSAGE_LENGTH)
275 {
276 err_code |= RD_ERROR_INVALID_PARAM;
277 }
278 else if (!command_is_authorized (raw_message[RE_STANDARD_OPERATION_INDEX]))
279 {
280 err_code |= reply_unauthorized (reply_fp, raw_message);
281 }
282 else
283 {
284 // Stop heartbeat processing.
285 err_code |= app_heartbeat_stop();
286#if APP_GATT_ENABLED
287 // Switch GATT to faster params.
288 err_code |= ri_gatt_params_request (RI_GATT_TURBO, CONN_PARAM_UPDATE_DELAY_MS);
289#endif
290 // Parse message type.
291 re_type_t type = raw_message[RE_STANDARD_DESTINATION_INDEX];
292
293 // Route message to proper handler.
294 switch (type)
295 {
296 case RE_ACC_XYZ:
297 case RE_ACC_X:
298 case RE_ACC_Y:
299 case RE_ACC_Z:
300 case RE_GYR_XYZ:
301 case RE_GYR_X:
302 case RE_GYR_Y:
303 case RE_GYR_Z:
304 case RE_ENV_ALL:
305 case RE_ENV_TEMP:
306 case RE_ENV_HUMI:
307 case RE_ENV_PRES:
308 err_code |= app_sensor_handle (reply_fp, raw_message, data_len);
309 break;
310
311 case RE_SEC_PASS:
312 err_code |= password_check (reply_fp, raw_message);
313 break;
314
315 default:
316 break;
317 }
318
319 // Switch GATT to slower params.
320#if APP_GATT_ENABLED
321 err_code |= ri_gatt_params_request (RI_GATT_LOW_POWER, 0);
322#endif
323 // Resume heartbeat processing.
324 err_code |= app_heartbeat_start();
325 }
326
327 RD_ERROR_CHECK (err_code, ~RD_ERROR_FATAL);
328}

711rd_status_t app_comms_blocking_send (const ri_comm_xfer_fp_t reply_fp,
712 ri_comm_message_t * const msg)
713{
714 rd_status_t err_code = RD_SUCCESS;
715 const uint64_t start = ri_rtc_millis();
716
717 do
718 {
