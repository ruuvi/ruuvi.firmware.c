88void setup (void)
89{

92# if (!RUUVI_RUN_TESTS)

95# endif
96 err_code |= protect_flash();
97 err_code |= ri_yield_init();
98 err_code |= ri_timer_init();
99 err_code |= ri_scheduler_init();
100 err_code |= rt_gpio_init();
101 err_code |= ri_yield_low_power_enable (true);
102 err_code |= rt_flash_init();
103 err_code |= app_led_init();
104 app_led_error_signal (true);
105 err_code |= app_button_init();
106 err_code |= app_dc_dc_init();
107 err_code |= app_sensor_init();
108 err_code |= app_log_init();
109 // Allow fail on boards which do not have accelerometer.
110 (void) app_sensor_acc_thr_set (&motion_threshold);
111 err_code |= app_comms_init (APP_LOCKED_AT_BOOT);
112 err_code |= app_sensor_vdd_sample();
113 err_code |= app_heartbeat_init();
114 err_code |= app_heartbeat_start();
115
116 if (RD_SUCCESS == err_code)
117 {
118 app_led_error_signal (false);
119 app_led_activity_signal (true);
120 err_code |= ri_delay_ms (APP_SELFTEST_OK_DELAY_MS);
121 app_led_activity_signal (false);
122 }
123
124 rd_error_cb_set (&app_on_error);
125 RD_ERROR_CHECK (err_code, RD_SUCCESS);
126}

129int app_main (void)
130#else
131int main (void)
132#endif
133{
134# if RUUVI_RUN_TESTS
135 integration_tests_run();
136# endif
137 setup();
138
139 do
140 {
141 ri_scheduler_execute();
142 // Sleep - woken up on event
143 // Do not indicate activity here to conserve power.
144 // Sensor reads take ~20ms, having led on for that time is expensive.
145 ri_yield();
146 // Prevent loop being optimized away

148 } while (LOOP_FOREVER);

150 // Intentionally non-reachable code unless unit testing.
