56app_dataformat_t app_dataformat_next (const app_dataformats_t formats,
57 const app_dataformat_t state)
58{
59 app_dataformat_t nextState = DF_INVALID;
60
61 if (DF_INVALID != formats.formats)
62 {
63 nextState = state;
64
65 do
66 {
67 switch (nextState)
68 {
69 case DF_3:
70 nextState = DF_5;
71 break;
72
73 case DF_5:
74 nextState = DF_8;
75 break;
76
77 case DF_8:
78 nextState = DF_C5;
79 break;
80
81 case DF_C5:
82 nextState = DF_FA;
83 break;
84
85 case DF_FA:
86 default:
87 nextState = DF_3;
88 break;
89 }
90 } while (! (nextState & formats.formats));
91 }
92
93 ri_adv_enable_uuid (nextState == DF_C5);
94 return nextState;
95}

168TESTABLE_STATIC rd_status_t
169ep_8_key_generate (uint8_t * const key)
170{
171 rd_status_t err_code = RD_SUCCESS;
172 memcpy (key, ep_8_key, RE_8_CIPHERTEXT_LENGTH);
173 uint64_t device_id = 0;
174 err_code |= ri_comm_id_get (&device_id);
175
176 for (uint8_t ii = 0U; ii < 8; ii++)
177 {
178 key[ii] = key[ii] ^ ( (device_id >> (ii * 8U)) & 0xFFU);
179 }
180
181 return err_code;
182}
