318 rd_status_t app_sensor_init (void)
319 {
320  rd_status_t err_code = RD_SUCCESS;
321  m_sensors_init();
322  err_code |= app_sensor_buses_init();
323 
324  if (RD_SUCCESS == err_code)
325  {
326  app_sensor_rtc_init();
327  // Wait for the power lines to settle after bus powerup.
328  ri_delay_ms (POWERUP_DELAY_MS);
329 
330  for (size_t ii = 0; ii < SENSOR_COUNT; ii++)
331  {
332  rd_status_t init_code = RD_SUCCESS;
333  size_t retries = 0;
334 
335  // Enable power to sensor
336  if (m_sensors[ii]->pwr_pin != RI_GPIO_ID_UNUSED)
337  {
338  (void) ri_gpio_configure (m_sensors[ii]->pwr_pin,
339  RI_GPIO_MODE_OUTPUT_HIGHDRIVE);
340  (void) ri_gpio_write (m_sensors[ii]->pwr_pin, m_sensors[ii]->pwr_on);
341  ri_delay_ms (POWERUP_DELAY_MS);
342  }
343 
344  // Some sensors, such as accelerometer may fail on user moving the board. Retry.
345  do
346  {
347  init_code = rt_sensor_initialize (m_sensors[ii]);
348  } while ( (APP_SENSOR_SELFTEST_RETRIES > retries++)
349 
350  && (RD_ERROR_SELFTEST == init_code));
351 
352  if (RD_SUCCESS == init_code)
353  {
354  // Check for a configuration in flash.
355  init_code = rt_sensor_load (m_sensors[ii]);
356 
357  // Configuration found, use it.
358  if (RD_SUCCESS == init_code)
359  {
360  init_code = rt_sensor_configure (m_sensors[ii]);
361  }
362  // Configuration not found, use defaults, store to flash.
363  else
364  {
365  init_code = rt_sensor_configure (m_sensors[ii]);
366  rt_sensor_store (m_sensors[ii]);
367  }
368  }
369  else if (RD_ERROR_SELFTEST == init_code)
370  {
371  err_code |= RD_ERROR_SELFTEST;
372  }
373  // Mark unavailable sensor handles as unused.
374  else
375  {
376  m_sensors[ii]->handle = APP_SENSOR_HANDLE_UNUSED;
377  }
378  }
379  }
380 
381  return err_code;
382 }

797 static rd_status_t app_sensor_log_read (const ri_comm_xfer_fp_t reply_fp,
798  const rd_sensor_data_fields_t fields,
799  const uint8_t * const raw_message)
800 {
801  rd_status_t err_code = RD_SUCCESS;
802  rd_sensor_data_t sample = {0};
803  sample.fields = fields;
804  float data[rd_sensor_data_fieldcount (&sample)];
805  sample.data = data;
806  // Parse start, end times.
807  int64_t current_time_s = (int64_t) re_std_log_current_time (raw_message);
808  int64_t start_s = (int64_t) re_std_log_start_time (raw_message);
809  uint32_t sent_elements = 0;
810  // overflow in 292 277 266 years
811  const int64_t system_time_ms = (int64_t) ri_rtc_millis();
812 
813  // Cannot have start_s >= current_time_s
814  if (current_time_s > start_s)
815  {
816  // Parse offset to system clock
817  LOG ("Sending logged data\r\n");
818  int64_t system_time_s = (system_time_ms / 1000LL);
819  int64_t offset_ms = (current_time_s - system_time_s) * 1000LL;
820  int64_t time_diff_ms = (current_time_s - start_s) * 1000LL;
821  // First sample to send in real time
822  sample.timestamp_ms = (start_s * 1000LL);
823 
824  // Offset sample time to system clock.
825  if (time_diff_ms > system_time_ms)
826  {
827  sample.timestamp_ms = 0;
828  }
829  else
830  {
831  sample.timestamp_ms = system_time_ms - time_diff_ms;
832  }
833 
834  app_log_read_state_t rs =
835  {
836  .oldest_element_ms = sample.timestamp_ms,
837  .element_idx = 0,
838  .page_idx = 0
839  };
840 
841  while (RD_SUCCESS == err_code)
842  {
843  // Reset data validity
844  sample.valid.bitfield = 0;
845  // Timestamp and fields are set in log read function.
846  err_code |= app_log_read (&sample, &rs);
847 
848  if (RD_ERROR_NOT_FOUND == err_code)
849  {
850  err_code |= app_sensor_send_eof (reply_fp, raw_message);
851  char msg[128];
852  snprintf (msg, sizeof (msg), "Logged data sent: %lu elements\r\n", sent_elements); //-V576
853  LOG (msg);
854  sent_elements = 0;
855  }
856  else if (app_heartbeat_overdue())
857  {
858  err_code |= RD_ERROR_TIMEOUT;
859  err_code |= app_sensor_send_timeout (reply_fp, raw_message);
860  }
861  // If data element was found, send log element.
862  else if (RD_SUCCESS == err_code)
863  {
864  err_code |= app_sensor_send_data (reply_fp, raw_message,
865  &sample, offset_ms);
866  sent_elements++;
867  LOGD ("S");
868  }
869  else
870  {
871  // No action needed, error code gets returned to caller.
872  }
873  }
874  }
875  // Start time >= current time
876  else
877  {
878  err_code |= RD_ERROR_INVALID_PARAM;
879  }
880 
881  // Send op status, remove expected not found
882  err_code &= ~RD_ERROR_NOT_FOUND;
883  return err_code;
884 }

886 rd_status_t app_sensor_handle (const ri_comm_xfer_fp_t reply_fp,
887  const uint8_t * const raw_message,
888  const uint16_t data_len)
889 {
890  rd_status_t err_code = RD_SUCCESS;
891 
892  if (NULL == raw_message)
893  {
894  err_code |= RD_ERROR_NULL;
895  }
896  else if (data_len < RE_STANDARD_MESSAGE_LENGTH)
897  {
898  err_code |= RD_ERROR_DATA_SIZE;
899  }
900  else
901  {
902  // Parse affected fields. It's ok to have unspecified value here,
903  // in that case fields end up being empty and error is reported via reply_fp.
904  re_type_t type = (re_type_t) raw_message[RE_STANDARD_DESTINATION_INDEX];
905  rd_sensor_data_fields_t target_fields = re2rd_fields (type);
906  // Parse desired operation.
907  re_op_t op = (re_op_t) raw_message[RE_STANDARD_OPERATION_INDEX];
908 
909  // If target and op are valid, execute.
910  switch (op)
911  {
912  case RE_STANDARD_LOG_VALUE_READ:
913  err_code |= app_sensor_log_read (reply_fp,
914  target_fields, raw_message);
915  break;
916 
917  default:
918  // Reply with error on unknown op.
919  break;
920  }
921  }
922 
923  return err_code;
924 }
