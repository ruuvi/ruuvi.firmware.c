326rd_status_t app_sensor_init (void)
327{
328 rd_status_t err_code = RD_SUCCESS;
329 m_sensors_init();
330 ri_i2c_frequency_t i2c_freq = rb_to_ri_i2c_freq (RB_I2C_FREQ);
331 // Initialize with slowest frequency supported by board to check all sensors
332 err_code |= app_sensor_buses_init (i2c_freq);
333 // Assume maximum speed board supports, if a sensor supports only lower speeds will get downgraded
334 i2c_freq = RB_I2C_MAX_SPD;
335
336 if (RD_SUCCESS == err_code)
337 {
338 app_sensor_rtc_init();
339 // Wait for the power lines to settle after bus powerup.
340 ri_delay_ms (POWERUP_DELAY_MS);
341
342 for (size_t ii = 0; ii < SENSOR_COUNT; ii++)
343 {
344 rd_status_t init_code = RD_SUCCESS;
345 size_t retries = 0;
346
347 // Enable power to sensor
348 if (m_sensors[ii]->pwr_pin != RI_GPIO_ID_UNUSED)
349 {
350 (void) ri_gpio_configure (m_sensors[ii]->pwr_pin,
351 RI_GPIO_MODE_OUTPUT_HIGHDRIVE);
352 (void) ri_gpio_write (m_sensors[ii]->pwr_pin, m_sensors[ii]->pwr_on);
353 ri_delay_ms (POWERUP_DELAY_MS);
354 }
355
356 // Some sensors, such as accelerometer may fail on user moving the board. Retry.
357 do
358 {
359 init_code = rt_sensor_initialize (m_sensors[ii]);
360 } while ( (APP_SENSOR_SELFTEST_RETRIES > retries++)
361
362 && (RD_ERROR_SELFTEST == init_code));
363
364 if (RD_SUCCESS == init_code)
365 {
366 // Check for a configuration in flash.
367 init_code = rt_sensor_load (m_sensors[ii]);
368
369 // Configuration found, use it.
370 if (RD_SUCCESS == init_code)
371 {
372 init_code = rt_sensor_configure (m_sensors[ii]);
373 }
374 // Configuration not found, use defaults, store to flash.
375 else
376 {
377 init_code = rt_sensor_configure (m_sensors[ii]);
378 rt_sensor_store (m_sensors[ii]);
379 }
380
381 // Update board max I2C speed
382 if (i2c_freq > m_sensors[ii]->i2c_max_speed)
383 {
384 i2c_freq = m_sensors[ii]->i2c_max_speed;
385 }
386 }
387 else if (RD_ERROR_SELFTEST == init_code)
388 {
389 err_code |= RD_ERROR_SELFTEST;
390 }
391 // Mark unavailable sensor handles as unused.
392 else
393 {
394 m_sensors[ii]->handle = APP_SENSOR_HANDLE_UNUSED;
395 }
396 }
397
398 // Reinit board with fastest speed supported by board + sensors.
399 err_code |= app_sensor_buses_uninit();
400 err_code |= app_sensor_buses_init (i2c_freq);
401 }
402
403 return err_code;
404}

406rd_status_t app_sensor_uninit (void)
407{
408 rd_status_t err_code = RD_SUCCESS;
409
410 for (size_t ii = 0; ii < SENSOR_COUNT; ii++)
411 {
412 if ( (NULL != m_sensors[ii]) && rd_sensor_is_init (& (m_sensors[ii]->sensor)))
413 {
414 m_sensors[ii]->sensor.uninit (&m_sensors[ii]->sensor, m_sensors[ii]->bus,
415 m_sensors[ii]->handle);
416
417 // Disable power to sensor
418 if (m_sensors[ii]->pwr_pin != RI_GPIO_ID_UNUSED)
419 {
420 (void) ri_gpio_write (m_sensors[ii]->pwr_pin, !m_sensors[ii]->pwr_on);
421 (void) ri_gpio_configure (m_sensors[ii]->pwr_pin, RI_GPIO_MODE_HIGH_Z);
422 }
423 }
424 }
425
426 err_code |= app_sensor_buses_uninit();
427 app_sensor_rtc_uninit();
428 ri_radio_activity_callback_set (NULL);
429 return err_code;
430}

488rd_status_t app_sensor_acc_thr_set (float * const threshold_g)
489{
490 rd_status_t err_code = RD_SUCCESS;
491 const rd_sensor_data_fields_t acceleration =
492 {
493 .datas.acceleration_x_g = 1,
494 .datas.acceleration_y_g = 1,
495 .datas.acceleration_z_g = 1
496 };
497 const rd_sensor_t * const provider = app_sensor_find_provider (acceleration);
498
499 if (RI_GPIO_ID_UNUSED == RB_INT_LEVEL_PIN)
500 {
501 err_code |= RD_ERROR_NOT_SUPPORTED;
502 }
503 else if ( (NULL == provider) || (NULL == provider->level_interrupt_set))
504 {
505 err_code |= RD_ERROR_NOT_SUPPORTED;
506 }
507 else if (NULL == threshold_g)
508 {
509 ri_gpio_interrupt_disable (RB_INT_LEVEL_PIN);
510 err_code |= provider->level_interrupt_set (false, threshold_g);
511 }
512 else if (0 > *threshold_g)
513 {
514 err_code |= RD_ERROR_NOT_IMPLEMENTED;
515 }
516 else
517 {
518 err_code |= ri_gpio_interrupt_enable (RB_INT_LEVEL_PIN,
519 RI_GPIO_SLOPE_TOGGLE,
520 RI_GPIO_MODE_INPUT_NOPULL,
521 &on_accelerometer_isr);
522 err_code |= provider->level_interrupt_set (true, threshold_g);
523 }
524
525 return err_code;
526}

608static uint8_t rd2re_fields (const rd_sensor_data_bitfield_t fields)
609{
610 // Implementing field->header assigments as a look-up table would
611 // rely on specific representation of bitfield at compile time.
612 // Little-endian, big-endian, BE-8, BE-32 etc. Using if-else
613 // here for robustness.
614 uint8_t header_value = 0;
615
616 if (fields.acceleration_x_g)
617 {
618 header_value = RE_ACC_X;
619 }
620 else if (fields.acceleration_y_g)
621 {
622 header_value = RE_ACC_Y;
623 }
624 else if (fields.acceleration_z_g)
625 {
626 header_value = RE_ACC_Z;
627 }
628 else if (fields.gyro_x_dps)
629 {
630 header_value = RE_GYR_X;
631 }
632 else if (fields.gyro_y_dps)
633 {
634 header_value = RE_GYR_Y;
635 }
636 else if (fields.gyro_z_dps)
637 {
638 header_value = RE_GYR_Z;
639 }
640 else if (fields.humidity_rh)
641 {
642 header_value = RE_ENV_HUMI;
643 }
644 else if (fields.pressure_pa)
645 {
646 header_value = RE_ENV_PRES;
647 }
648 else if (fields.temperature_c)
649 {
650 header_value = RE_ENV_TEMP;
651 }
652 else
653 {
654 // No action needed
655 }
656
657 return header_value;

739static rd_status_t app_sensor_send_data (const ri_comm_xfer_fp_t reply_fp,
740 const uint8_t * const raw_message,
741 const rd_sensor_data_t * const sample,
742 const int64_t time_offset_ms)
743{
744 rd_status_t err_code = RD_SUCCESS;
745 const uint8_t fieldcount = rd_sensor_data_fieldcount (sample);
746 int64_t real_time_ms = 0;
747
748 if ( (0 - time_offset_ms) < (int64_t) sample->timestamp_ms)
749 {
750 real_time_ms = sample->timestamp_ms + time_offset_ms;
751 }
752
753 for (uint8_t ii = 0; ii < fieldcount; ii++)
754 {
755 if (rd_sensor_has_valid_data (sample, ii))
756 {
757 rd_sensor_data_bitfield_t type = rd_sensor_field_type (sample, ii);
758 err_code |= send_field (reply_fp, raw_message,
759 type, sample->data[ii], real_time_ms);
760 }
761 }
762
763 return err_code;
764}

819static rd_status_t app_sensor_log_read (const ri_comm_xfer_fp_t reply_fp,
820 const rd_sensor_data_fields_t fields,
821 const uint8_t * const raw_message)
822{
823 rd_status_t err_code = RD_SUCCESS;
824 rd_sensor_data_t sample = {0};
825 sample.fields = fields;
826 float data[rd_sensor_data_fieldcount (&sample)];
827 sample.data = data;
828 // Parse start, end times.
829 int64_t current_time_s = (int64_t) re_std_log_current_time (raw_message);
830 int64_t start_s = (int64_t) re_std_log_start_time (raw_message);
831 uint32_t sent_elements = 0;
832 // overflow in 292 277 266 years
833 const int64_t system_time_ms = (int64_t) ri_rtc_millis();
834
835 // Cannot have start_s >= current_time_s
836 if (current_time_s > start_s)
837 {
838 // Parse offset to system clock
839 LOG ("Sending logged data\r\n");
840 int64_t system_time_s = (system_time_ms / 1000LL);
841 int64_t offset_ms = (current_time_s - system_time_s) * 1000LL;
842 int64_t time_diff_ms = (current_time_s - start_s) * 1000LL;
843 // First sample to send in real time
844 sample.timestamp_ms = (start_s * 1000LL);
845
846 // Offset sample time to system clock.
847 if (time_diff_ms > system_time_ms)
848 {
849 sample.timestamp_ms = 0;
850 }
851 else
852 {
853 sample.timestamp_ms = system_time_ms - time_diff_ms;
854 }
855
856 app_log_read_state_t rs =
857 {
858 .oldest_element_ms = sample.timestamp_ms,
859 .element_idx = 0,
860 .page_idx = 0
861 };
862
863 while (RD_SUCCESS == err_code)
864 {
865 // Reset data validity
866 sample.valid.bitfield = 0;
867 // Timestamp and fields are set in log read function.
868 err_code |= app_log_read (&sample, &rs);
869
870 if (RD_ERROR_NOT_FOUND == err_code)
871 {
872 err_code |= app_sensor_send_eof (reply_fp, raw_message);
873 char msg[128];
874 snprintf (msg, sizeof (msg), "Logged data sent: %lu elements\r\n", sent_elements); //-V576
875 LOG (msg);
876 sent_elements = 0;
877 }
878 else if (app_heartbeat_overdue())
879 {
880 err_code |= RD_ERROR_TIMEOUT;
881 err_code |= app_sensor_send_timeout (reply_fp, raw_message);
882 }
883 // If data element was found, send log element.
884 else if (RD_SUCCESS == err_code)
885 {
886 err_code |= app_sensor_send_data (reply_fp, raw_message,
887 &sample, offset_ms);
888 sent_elements++;
889 LOGD ("S");
890 }
891 else
892 {
893 // No action needed, error code gets returned to caller.
894 }
895 }
896 }
897 // Start time >= current time
898 else
899 {
900 err_code |= RD_ERROR_INVALID_PARAM;
901 }
902
903 // Send op status, remove expected not found
904 err_code &= ~RD_ERROR_NOT_FOUND;
905 return err_code;
906}

908rd_status_t app_sensor_handle (const ri_comm_xfer_fp_t reply_fp,
909 const uint8_t * const raw_message,
910 const uint16_t data_len)
911{
912 rd_status_t err_code = RD_SUCCESS;
913
914 if (NULL == raw_message)
915 {
916 err_code |= RD_ERROR_NULL;
917 }
918 else if (data_len < RE_STANDARD_MESSAGE_LENGTH)
919 {
920 err_code |= RD_ERROR_DATA_SIZE;
921 }
922 else
923 {
924 // Parse affected fields. It's ok to have unspecified value here,
925 // in that case fields end up being empty and error is reported via reply_fp.
926 re_type_t type = (re_type_t) raw_message[RE_STANDARD_DESTINATION_INDEX];
927 rd_sensor_data_fields_t target_fields = re2rd_fields (type);
928 // Parse desired operation.
929 re_op_t op = (re_op_t) raw_message[RE_STANDARD_OPERATION_INDEX];
930
931 // If target and op are valid, execute.
932 switch (op)
933 {
934 case RE_STANDARD_LOG_VALUE_READ:
935 err_code |= app_sensor_log_read (reply_fp,
936 target_fields, raw_message);
937 break;
938
939 default:
940 // Reply with error on unknown op.
941 break;
942 }
943 }
944
945 return err_code;
946}
