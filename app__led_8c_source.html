37static void state_change_process (void)
38{
39 rd_status_t err_code = RD_SUCCESS;
40
41 //State: Paused
42 if (PAUSE_BIT & m_signals)
43 {
44 // No action needed
45 }
46 //State: Error on
47 else if (ERROR_BIT & m_signals)
48 {
49 err_code |= rt_led_write (RB_LED_STATUS_ERROR, true);
50 }
51 //State: Interaction on
52 else if (INTERACTION_BIT & m_signals)
53 {
54 err_code |= rt_led_write (RB_LED_BUTTON_PRESS, true);
55 }
56 // State: configurable, active
57 else if ( (ACTIVITY_BIT | CONFIGURABLE_BIT) == m_signals)
58 {
59 err_code |= rt_led_write (RB_LED_CONFIG_ENABLED, true);
60 }
61 // State: active, not configurable
62 else if (ACTIVITY_BIT == m_signals)
63 {
64 err_code |= rt_led_write (RB_LED_ACTIVITY, true);
65 }
66 //Default: leds off
67 else
68 {
69 for (size_t ii = 0; ii < RB_LEDS_NUMBER; ii++)
70 {
71 err_code |= rt_led_write (m_led_pins[ii], false);
72 }
73 }
74
75 RD_ERROR_CHECK (err_code, ~RD_ERROR_FATAL);
76}
