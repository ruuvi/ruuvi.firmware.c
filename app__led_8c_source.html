38static void state_change_process (void)
39{
40 rd_status_t err_code = RD_SUCCESS;
41 const bool error_led_active = ERROR_BIT & m_signals;
42 const bool motion_led_active = MOTION_BIT & m_signals;
43 const bool presence_led_active = PRESENCE_BIT & m_signals;
44 const bool activity_led_active = ACTIVITY_BIT & m_signals;
45 const bool interaction_led_active = INTERACTION_BIT & m_signals;
46
47 //State: Paused
48 if (PAUSE_BIT & m_signals)
49 {
50 return;
51 }
52
53 //Default: leds off
54 for (uint32_t ii = 0; ii < RB_LEDS_NUMBER; ii++)
55 {
56 err_code |= rt_led_write (m_led_pins[ii], false);
57 }
58
59 if (error_led_active)
60 {
61 err_code |= rt_led_write (RB_LED_STATUS_ERROR, true);
62 }
63
64 if (interaction_led_active)

67 }
68

73
74 if (presence_led_active)

76 err_code |= rt_led_write (RB_LED_PRESENCE, true);
77 }
78
79 if (activity_led_active)
80 {
81 // Route activity to signal function
82 err_code |= app_led_activity_indicate (true);
83 }
